#!/bin/sh

# Show dialog with message and a single OK button.
# New lines and multiple spaces inside arguments are preserved.
# Formatting like bold, underline, and colors is supported.
# See `man 1 dialog` for formatting syntax.
debug_dialog(){
    dialog \
        --colors \
        --no-collapse \
        --title  "Felipe: DEBUG" \
        --msgbox "$1" \
        0 0
}

debug_dialog "hello dialog"
